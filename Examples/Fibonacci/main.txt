// Generate the Fibonacci series up to num_elements

int num_elements
int f[100]

assign f[0]: 0
assign f[1]: 1

read num_elements

for int i: 0; i <= num_elements; assign i: i + 1 {
    if i >= 2 {
        assign f[i]: f[i - 1] + f[i - 2]
    }
    print "f["
    print i
    print "] is "
    print f[i]
    print "endl"
}

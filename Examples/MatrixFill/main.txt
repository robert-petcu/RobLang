// Generate a square matrix of dimension N, with values 1, 2, 3, ..., N, ..., N * N

int dimension
int a[1000][1000]

read dimension

for int i: 1; i <= dimension; assign i: i + 1 {
    for int j: 1; j <= dimension; assign j: j + 1 {
        assign a[i][j]: (i - 1) * dimension + j
        print a[i][j]
        print " "
    }
    print "endl"
}
